function Ninja(name){ var speed = 3; var strength = 3; this.name = name; this.health = 100; this.sayName = function(){ console.log(name); } this.showStats = function(){ console.log(`Speed: ${speed} Strength: ${strength} Health: ${this.health}`); } this.drinkSake = function(){ this.health += 10; } this.punch = function(otherNinja){ console.log(`Original Health of ${otherNinja.name}:`); otherNinja.showStats(); console.log(`${otherNinja.name} was punched by ${this.name} and lost 5 Health`); otherNinja.health -= 5; otherNinja.showStats(); } this.kick = function(otherNinja){ console.log(`Original Health of ${this.name}:`); this.showStats(); console.log(`${this.name} was kicked by ${otherNinja.name} and lost 15 Health`); this.health -= 15; console.log(`After getting kicked, Health of ${this.name}:`); this.showStats(); } } var ninja1 = new Ninja("Hyabusa"); var ninja2 = new Ninja("Goemon"); ninja1.punch(ninja2); ninja1.kick(ninja2);
